import serial
import socket

ser = serial.Serial("COM3", 9600, timeout=1)
sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)

while True:
    line = ser.readline().decode("ascii", errors="ignore").strip()
    if line:
        print("Sending:", line)
        sock.sendto(line.encode(), ("127.0.0.1", 4242))
